  function parseDef(text) {
    var m = text.match(/^\[([^\]]+)\]/);
    if (!m) return text;
    var tags = m[1].split(','), html = '', ok = true;
    for (var i = 0; i < tags.length; i++) {
      var t = trim(tags[i]);
      if (registers[t])     html += '<span class="register-tag ' + registers[t] + '">' + t + '</span>';
      else if (subjects[t]) html += '<span class="register-tag rt-subject">' + t + '</span>';
      else { ok = false; break; }
    }
    return (ok && html) ? html + ' ' + trim(text.substring(m[0].length)) : text;
  }

  function splitDefinitionGloss(text) {
    var value = trim(text);
    if (!value || value.charAt(value.length - 1) !== ')') return {en:value, vi:''};
    var depth = 0;
    for (var i = value.length - 1; i >= 0; i--) {
      var ch = value.charAt(i);
      if (ch === ')') depth++;
      else if (ch === '(') {
        depth--;
        if (depth === 0) {
          var en = trim(value.substring(0, i));
          var vi = trim(value.substring(i + 1, value.length - 1));
          return (en && vi) ? {en:en, vi:vi} : {en:value, vi:''};
        }
        if (depth < 0) return {en:value, vi:''};
      }
    }
    return {en:value, vi:''};
  }

    for (var p = 0; p < parts.length; p++) {
      if (parts[p].label === 'UK') {
        html += '<span class="ipa-chip ipa-chip-uk"><span class="ipa-label">UK</span><span class="ipa-symbols">' + escapeHtml(parts[p].value) + '</span></span>';
      } else {
        html += '<span class="ipa-chip ipa-chip-us"><span class="ipa-label">US</span><span class="ipa-symbols">' + escapeHtml(parts[p].value) + '</span></span>';
      }
    }
    return html + '</span>';
  }

  // 1. POS chips
  var rawPos = getRaw('raw-pos-back');
  var rawPosUrls = getRaw('raw-oxford-pos-urls-back');
  var posDiv = document.getElementById('top-bar-pos-back');
  var topBarLeft = posDiv ? posDiv.parentNode : null;
  renderPosChips(posDiv, rawPos, rawPosUrls);

  // 2. Corpus badges + separator between POS and corpus
  var rawTags = getRaw('raw-tags-back');
  var corpusMap = {
    'Oxford_3000':['corpus-badge corpus-oxf','Oxford 3000'],
    'Oxford_5000':['corpus-badge corpus-oxf','Oxford 5000'],
    'OPAL_W':     ['corpus-badge corpus-opal','OPAL W'],
    'OPAL_S':     ['corpus-badge corpus-opal','OPAL S'],
    'AWL_Coxhead':['corpus-badge corpus-awl','AWL Coxhead']
  };
  var addedSep = false;
  if (topBarLeft && rawTags) {
    var tList = rawTags.split(/\s+/);
    for (var k = 0; k < tList.length; k++) {
      var ct = trim(tList[k]);
      if (corpusMap[ct]) {
        if (!addedSep) {
          var sep = document.createElement('span');
          sep.className = 'top-bar-sep';
          topBarLeft.appendChild(sep);
          addedSep = true;
        }
        var cs = document.createElement('span');
        cs.className = corpusMap[ct][0];
        cs.textContent = corpusMap[ct][1];
        topBarLeft.appendChild(cs);
      }
    }
  }

  // 3. Usage restriction tags → meta-row
  var usageContainer = document.getElementById('usage-tags-container');
  var usageMap = {
    'old_fashioned':'old-fashioned','old_use':'old use',
    'dialect':'dialect','saying':'saying','trademark':'\u2122 trademark'
  };
  if (usageContainer && rawTags) {
    var uHtml = '';
    var uList = rawTags.split(/\s+/);
    for (var u = 0; u < uList.length; u++) {
      var ut = trim(uList[u]);
      if (usageMap[ut]) uHtml += '<span class="usage-tag">' + usageMap[ut] + '</span>';
    }
    usageContainer.innerHTML = uHtml;
  }

  var ipaEl = document.getElementById('ipa-display');
  if (ipaEl) ipaEl.innerHTML = renderIpa(ipaEl.textContent || ipaEl.innerText || '');

  var audioRow = document.getElementById('audio-row-container');
  if (audioRow && !trim(audioRow.textContent || audioRow.innerText)) {
    audioRow.style.display = 'none';
  }

  // 4. L3 Senses grid
  var rawDef  = getRaw('raw-definition-back');
  var rawDefVi = getRaw('raw-definition-vi-back');
  var rawEx   = getRawExampleHtml('raw-example-back');
  var rawWord = getRaw('raw-word-back');
  var rawSyns = getRaw('raw-synonyms-back');
  var rawAnts = getRaw('raw-antonyms-back');
  var mainAudioUk = parseMainAudio(getRawAudioHtml('raw-example-audio-uk-back'));
  var mainAudioUs = parseMainAudio(getRawAudioHtml('raw-example-audio-us-back'));
  var sensesEl = document.getElementById('senses-container');

  if (sensesEl && rawDef) {
    var defParts = rawDef.split('|'), defViParts = rawDefVi ? rawDefVi.split('|') : [];
    var exParts = rawEx ? rawEx.split('|') : [];
    var synParts = rawSyns ? rawSyns.split('|') : [];
    var antParts = rawAnts ? rawAnts.split('|') : [];
    var defs = [];
    for (var d = 0; d < defParts.length; d++) { if (trim(defParts[d])) defs.push({i:d, t:trim(defParts[d])}); }

    var sHtml = '';

    // Render synonym/antonym parentheticals with their colored class.
    // Wrap each paren when all its items belong to one relation channel.
    // A channel may be the union of relations shown across multiple examples in the same sense.
    // Supports single words `(word)`, `(= word)` and grouped lists `(w1, w2)`, `(= w1, w2)`.
    function renderRelations(exHtml, synWords, antWords) {
      if (!exHtml) return exHtml;
      var synList = [];
      if (synWords) {
        for (var s = 0; s < synWords.length; s++) {
          var sw = trim(synWords[s]).toLowerCase();
          if (sw && synList.indexOf(sw) === -1) synList.push(sw);
        }
      }
      var antList = [];
      if (antWords) {
        for (var a = 0; a < antWords.length; a++) {
          var aw = trim(antWords[a]).toLowerCase();
          if (aw && antList.indexOf(aw) === -1) antList.push(aw);
        }
      }
      if (!synList.length && !antList.length) return exHtml;

      return exHtml.replace(/\(([^()]*)\)/g, function(match, inner) {
        var trimmed = trim(inner).replace(/^=\s*/, '').trim();
        if (!trimmed) return match;
        var parts = trimmed.split(',');
        var cleanSet = [];
        for (var i = 0; i < parts.length; i++) {
          var item = trim(parts[i]).toLowerCase();
          if (!item || cleanSet.indexOf(item) !== -1) return match;
          cleanSet.push(item);
        }
        if (!cleanSet.length) return match;

        var isSynSubset = synList.length > 0;
        if (isSynSubset) {
          for (var j = 0; j < cleanSet.length; j++) {
            if (synList.indexOf(cleanSet[j]) === -1) {
              isSynSubset = false;
              break;
            }
          }
        }

        var isAntSubset = antList.length > 0;
        if (isAntSubset) {
          for (var k = 0; k < cleanSet.length; k++) {
            if (antList.indexOf(cleanSet[k]) === -1) {
              isAntSubset = false;
              break;
            }
          }
        }

        if (isSynSubset && isAntSubset) return match;
        if (isSynSubset) return '<span class="relation-synonym">' + match + '</span>';
        if (isAntSubset) return '<span class="relation-antonym">' + match + '</span>';
        return match;
      });
    }

    for (var di = 0; di < defs.length; di++) {
      var idx = defs[di].i;
      var def = defs[di].t;
      var defPair = splitDefinitionGloss(def);
      var defVi = (idx < defViParts.length) ? trim(defViParts[idx]) : '';
      if (!defVi) defVi = defPair.vi;
      var ex  = (idx < exParts.length) ? trim(exParts[idx]) : '';
      var synWords = (idx < synParts.length) ? trim(synParts[idx]).split(',') : [];
      var antWords = (idx < antParts.length) ? trim(antParts[idx]).split(',') : [];
      for (var sw = 0; sw < synWords.length; sw++) synWords[sw] = trim(synWords[sw]);
      for (var aw = 0; aw < antWords.length; aw++) antWords[aw] = trim(antWords[aw]);
      // Only render relations if the per-chunk metadata has entries;
      // otherwise leave the example untouched (no inference from
      // sense-level metadata without per-chunk evidence).
      var hasRelationMeta = false;
      for (var sw2 = 0; sw2 < synWords.length; sw2++) { if (synWords[sw2]) { hasRelationMeta = true; break; } }
      if (!hasRelationMeta) {
        for (var aw2 = 0; aw2 < antWords.length; aw2++) { if (antWords[aw2]) { hasRelationMeta = true; break; } }
      }
      var exHtml = '';
      if (ex) {
        var exLines = ex.split(/(?:<br\s*\/?>){2}/i);
        for (var el = 0; el < exLines.length; el++) {
          var exLine = trim(exLines[el]);
          if (!exLine) continue;
          exLine = highlightHeadword(exLine, rawWord, rawPos);
          if (hasRelationMeta) exLine = renderRelations(exLine, synWords, antWords);
          var ukSource = (idx < mainAudioUk.length && el < mainAudioUk[idx].length) ? mainAudioUk[idx][el] : '';
          var usSource = (idx < mainAudioUs.length && el < mainAudioUs[idx].length) ? mainAudioUs[idx][el] : '';
          exHtml += exampleAudioLine('\u201c' + exLine + '\u201d', ukSource, usSource,
                                    'sense ' + (di + 1) + ' example ' + (el + 1), '');
        }
      }
      var viHtml = defVi ? '<div class="sense-vi" lang="vi" aria-label="Vietnamese gloss"><span class="sense-vi-label" aria-hidden="true">VI</span><span class="sense-vi-text">' + escapeHtml(defVi) + '</span></div>' : '';
      sHtml += '<div class="sense-row">'
             + '<div class="sense-def"><div class="sense-en">' + parseDef(defPair.en) + '</div>' + viHtml + '</div>'
             + '<div class="sense-ex">' + exHtml + '</div>'
             + '</div>';
    }
    sensesEl.innerHTML = sHtml;
  }

  // 5. Word family chips
  var wf = getRaw('raw-word-family-back');
  var wfEl = document.getElementById('word-family-container');
  if (wf && wfEl) {
    var posMap = {'n':'wf-pos-n','v':'wf-pos-v','adj':'wf-pos-adj','adv':'wf-pos-adv','phr':'wf-pos-phr','prep':'wf-pos-prep'};
    var lines = wf.split(/\\\\n|\\n|\n/);
    var wHtml = '';
    for (var r = 0; r < lines.length; r++) {
      var line = trim(lines[r]);
      if (!line) continue;
      var wm = line.match(/^(.+?)\s*\(([^)]+)\)$/);
      if (wm) {
        var wd = trim(wm[1]), ps = trim(wm[2]).toLowerCase();
        var pc = posMap[ps] || 'wf-pos-n';
        wHtml += '<span class="wf-chip"><span class="wf-chip-word">' + wd + '</span><span class="wf-chip-pos ' + pc + '">' + ps + '</span></span>';
      } else {
        wHtml += '<span class="wf-chip"><span class="wf-chip-word">' + line + '</span></span>';
      }
    }
    wfEl.innerHTML = wHtml;
  }

  // 6. Collocations chips
  var rawCol = getRaw('raw-collocations-back');
  var rawColSources = getRaw('raw-collocation-sources-back');
  var colEl  = document.getElementById('collocations-container');
  if (rawCol && colEl) {
    renderCollocations(rawCol, rawColSources, colEl);
  }

  // 7. Feature tags → feature-row
  var featureEl = document.getElementById('feature-row-container');
  if (featureEl && rawTags) {
    var fHtml = '';
    var fList = rawTags.split(/\s+/);
    for (var f = 0; f < fList.length; f++) {
      var ft = trim(fList[f]);
      if (ft === 'idioms')         fHtml += '<span class="feature-tag"><i class="ti ti-bookmarks"></i> idioms</span>';
      else if (ft === 'phrasal_verbs') fHtml += '<span class="feature-tag"><i class="ti ti-bookmarks"></i> phrasal verbs</span>';
    }
    if (fHtml) { featureEl.innerHTML = fHtml; featureEl.style.display = 'flex'; }
  }

  // 8. Idioms parser
  // Idioms: phrase :: English meaning :: ex1 | ex2 $$ phrase2 :: meaning :: ex1
  // IdiomMeaningVI: mode :: Vietnamese meaning $$ mode :: Vietnamese meaning
  var rawIdioms = getRaw('raw-idioms-back');
  var rawIdiomMeaningVi = getRaw('raw-idiom-meaning-vi-back');
  var idiomAudioUk = parseIdiomAudio(getRawAudioHtml('raw-idiom-example-audio-uk-back'));
  var idiomAudioUs = parseIdiomAudio(getRawAudioHtml('raw-idiom-example-audio-us-back'));
  var idiomsEl = document.getElementById('idioms-container');
  if (rawIdioms && idiomsEl) {
    idiomsEl.innerHTML = renderIdioms(rawIdioms, rawIdiomMeaningVi, idiomAudioUk, idiomAudioUs);
  }

  wireExampleAudio();

})();
</script>
