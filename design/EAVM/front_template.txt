<script>
(function() {
  function trim(s) { return s ? s.replace(/^\s+|\s+$/g, '') : ''; }
  function getRaw(id) { var e = document.getElementById(id); return e ? trim(e.textContent || e.innerText) : ''; }
  function renderPosChips(posDiv, rawPos, rawUrls) {
    if (!posDiv || !rawPos) return;
    var parts = rawPos.split(/[,\/]/), clean = [];
    for (var i = 0; i < parts.length; i++) {
      var part = trim(parts[i]);
      if (part) clean.push(part);
    }
    var urls = rawUrls ? rawUrls.split('|') : [];
    posDiv.textContent = '';
    for (var j = 0; j < clean.length; j++) {
      var url = trim(urls[j] || '');
      var chip = document.createElement(url ? 'a' : 'span');
      chip.className = 'pos-chip';
      if (url) {
        chip.href = url;
        chip.target = '_blank';
        chip.rel = 'noopener noreferrer';
        chip.setAttribute('aria-label', 'Open Oxford entry for ' + clean[j]);
      }
      if (clean.length > 1) {
        var number = document.createElement('span');
        number.className = 'pos-chip-num';
        number.textContent = j + 1;
        chip.appendChild(number);
      }
      chip.appendChild(document.createTextNode(clean[j]));
      posDiv.appendChild(chip);
    }
  }

  // 1. POS chips
  var rawPos = getRaw('raw-pos-front');
  var rawPosUrls = getRaw('raw-oxford-pos-urls-front');
  var posDiv = document.getElementById('top-bar-pos-front');
  var topBarLeft = posDiv ? posDiv.parentNode : null;
  renderPosChips(posDiv, rawPos, rawPosUrls);

  // 2. Corpus badges + separator
  var rawTags = getRaw('raw-tags-front');
  var corpusMap = {
    'Oxford_3000':['corpus-badge corpus-oxf','Oxford 3000'],
    'Oxford_5000':['corpus-badge corpus-oxf','Oxford 5000'],
    'OPAL_W':     ['corpus-badge corpus-opal','OPAL W'],
    'OPAL_S':     ['corpus-badge corpus-opal','OPAL S'],
    'AWL_Coxhead':['corpus-badge corpus-awl','AWL Coxhead']
  };
  var addedSep = false;
  if (topBarLeft && rawTags) {
    var tList = rawTags.split(/\s+/);
    for (var k = 0; k < tList.length; k++) {
      var ct = trim(tList[k]);
      if (corpusMap[ct]) {
        if (!addedSep) {
          var sep = document.createElement('span');
          sep.className = 'top-bar-sep';
          topBarLeft.appendChild(sep);
          addedSep = true;
        }
        var cs = document.createElement('span');
        cs.className = corpusMap[ct][0];
        cs.textContent = corpusMap[ct][1];
        topBarLeft.appendChild(cs);
      }
    }
  }

  // 3. Sense count dots (only if multi-sense)
  var rawDef = getRaw('raw-definition-front');
  var dotsEl = document.getElementById('sense-count-front');
  if (rawDef && dotsEl) {
    var parts2 = rawDef.split('|'), count = 0;
    for (var d = 0; d < parts2.length; d++) { if (trim(parts2[d])) count++; }
    if (count > 1) {
      var dots = '';
      for (var n = 0; n < count; n++) dots += '<span class="sense-count-dot"></span>';
      dotsEl.innerHTML = dots;
    }
  }

})();
</script>
